# Book-Reviews
This is book reviews app with searching and filtering books and adding review.

# step 1

copy .env.example .env

# step 2

composer install

# step 3

./vendor/bin/sail  up -d

# step 4

./vendor/bin/sail  artisan key:generate

# step 5

./vendor/bin/sail  artisan migrate

# step 6

sail npm install

# step 7

sail npm run dev


# step 8

http://localhost
